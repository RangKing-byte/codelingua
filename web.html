HTML 

<!-- Қарапайым HTML құрылымы -->
<!DOCTYPE html>
<html>
<head>
    <title>Менің веб-сайтым</title>
    <link rel="stylesheet" 
          href="style.css">
</head>
<body>
    <h1>Сәлем, Әлем!</h1>
    <p>Бұл менің веб-сайтым.</p>
</body>
</html>
